bool ismiddle(int arr[], int n){
int mid=arr[n/2]; int count=0;
for(int i=0; i<n;i++){
if(arr[i]==mid) count++;
}
if(count>=2) return true;
else return false;
}

int main() {
    int n, m;

    cin >> n >> m;

    int arr[n][m];
    int sum = 0;

    for(int i = 0; i < n; i++) {
        for(int j = 0; j < m; j++) {
            cin >> arr[i][j];
        }
    }

    for(int i = 0; i < n; i++) {
        for(int j = 0; j < m; j++) {
            sum += arr[i][j];
        }
    }

    cout << "Total Sum = " << sum;

    return 0;
}
