int main()
{
   //bubble sort
    int arr[] = {20,15, 10,5};
    int n = 4;
    for (int i = 0; i < n; i++) cout << arr[i] << " ";
    for (int i = 0; i < n - 1; i++) {
        for (int j = 0; j < n - 1-i; j++) {
            if (arr[j] < arr[j + 1])
            //if (arr[j] > arr[j + 1])
                swap(arr[j], arr[j + 1]);
        }
        cout << endl;
        for (int i = 0; i < n; i++) cout << arr[i] << " ";
        //selection  sort 
        for(int i=0; i<n-1;i++){       
            int min=arr[i];
            int mindx=i;
            for(int j=i+1;j<n;j++){
                if(arr[j]<min){
                    min=arr[j];
                    mindx=j;
                }
            }
            swap(arr[i],arr[mindx]);
        }
        //insertion sort
        for(int i=1; i<n; i++){
            int curr=arr[i]; int j=i-1;
            while(j>=0){
                if(arr[j]>curr) arr[j+1]=arr[j];
                else break;
                j--;
            }
            arr[j+1]=curr;
        }
        
    }
}

int main() {
    int arr1[] = {1, 2, 3};
    int arr2[] = {2, 4, 6};

    int n1 = 3, n2 = 3;
    int arr3[n1 + n2];

    
    for(int i = 0; i < n1; i++) {
        arr3[i] = arr1[i];
    }

    for(int i = 0; i < n2; i++) {
        arr3[n1 + i] = arr2[i];
    }

  
    sort(arr3, arr3 + n1 + n2);

    
    for(int i = 0; i < n1 + n2; i++) {
        cout << arr3[i] << " ";
    }

    return 0;
}
