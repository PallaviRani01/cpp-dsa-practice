#include <bits/stdc++.h>
using namespace std;

bool ratInMaze(vector<vector<int>> &maze, vector<vector<int>> &path, int i, int j) {

    int n = maze.size();
    int m = maze[0].size();

    if (i < 0 || j < 0 || i >= n || j >= m || maze[i][j] == 0 || path[i][j] == 1)
        return false;

    if (i == n - 1 && j == m - 1) {
        path[i][j] = 1;
        return true;
    }

    path[i][j] = 1;

    if (ratInMaze(maze, path, i - 1, j))
        return true;

    if (ratInMaze(maze, path, i + 1, j))
        return true;

    if (ratInMaze(maze, path, i, j + 1))
        return true;

    if (ratInMaze(maze, path, i, j - 1))
        return true;

    // Backtrack
    path[i][j] = 0;

    return false;
}

int main() {

    int n, m;
    cin >> n >> m;

    vector<vector<int>> maze(n, vector<int>(m));
    vector<vector<int>> path(n, vector<int>(m, 0));

    for (int i = 0; i < n; i++) {
        for (int j = 0; j < m; j++) {
            cin >> maze[i][j];
        }
    }

    if (ratInMaze(maze, path, 0, 0)) {

        cout << "Path Found\n";

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                cout << path[i][j] << " ";
            }
            cout << endl;
        }

    } else {
        cout << "No Path Found";
    }

    return 0;
}
