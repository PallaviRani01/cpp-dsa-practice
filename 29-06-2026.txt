#include <iostream>
using namespace std;

class Account {
public:
    int balance;

    // Constructor
    Account(int b) {
        balance = b;
    }
};

int main() {

    // Static Object
    Account obj1(5000);

    // Dynamic Object
    Account *obj2 = new Account(10000);

    // Print Balance
    cout << "Balance of Static Object: " << obj1.balance << endl;
    cout << "Balance of Dynamic Object: " << obj2->balance << endl;

    // Free Memory
    delete obj2;

    return 0;
}
