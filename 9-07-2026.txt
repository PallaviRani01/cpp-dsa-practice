#include <iostream>
using namespace std;

class Node {
public:
    int data;
    Node* next;

    Node(int data) {
        this->data = data;
        next = NULL;
    }
};
//node 
class Stack {
public:
    Node* top;

    Stack() {
        top = NULL;
    }
//Inset 

    void push(int x) {
        Node* newNode = new Node(x);
        newNode->next = top;
        top = newNode;
    }
//delete
    void pop() {
        if (top == NULL) {
            cout << "Stack Underflow\n";
            return;
        }

        Node* temp = top;
        top = top->next;
        delete temp;
    }
//top element
    int peek() {
        if (top == NULL)
            return -1;

        return top->data;
    }
// empty or not?
    bool empty() {
        return top == NULL;
    }
};

int main() {
    Stack s;

    s.push(10);
    s.push(20);
    s.push(30);

    cout << "Top = " << s.peek() << endl;

    s.pop();

    cout << "Top after pop = " << s.peek() << endl;
}

int main() {
    string s="aba";
    cin >> s;

    int i = 0;
    int j = s.length() - 1;

    bool palindrome = true;

    while (i < j) {
        if (s[i] != s[j]) {
            palindrome = false;
            break;
        }
        i++;
        j--;
    }

    if (palindrome)
        cout << "true";
    else
        cout << "false";

    return 0;
}
